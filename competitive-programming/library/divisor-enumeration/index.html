約数列挙

更新日：2019-10-29

引数の約数を列挙する。 ソートはされていない。

計算量：\(O(\sqrt{N})\)

template <typename T> vector<T> divisor(T N) {
    vector<T> ret;
    if (N == 1) {
        ret.emplace_back(1);
        return ret;
    }
    if(N <= 0) {
        return ret;
    }
    for (unsigned long long i = 1; i * i <= N; i++) {
        if (N % i == 0) {
            ret.emplace_back(i);
            if (i != 1 && i * i != N) {
                ret.emplace_back(N / i);
            }
        }
    }
    ret.emplace_back(N);
    return ret;
}
