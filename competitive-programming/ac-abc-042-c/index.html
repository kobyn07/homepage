AtCoderABCシミュレーション

問題文

方針

\(D_i\) の数字が出てこない \(N\) 以上の数を求めたい。

制約から 888,888 円以上は払う必要がない。 また、最大の桁数が 6 桁なので、ある数字が支払えるかどうか十分高速に調べられる。 よって、愚直に \(N\) 以上 888,888 円以下の数字で満たされている数字を毎回判定すればよい。

解法

#include <bits/stdc++.h>
using namespace std;

int main() {
    int N, K;
    cin >> N >> K;
    vector<int> D(K);
    for (auto &&i : D) {
        cin >> i;
    }

    auto check = [&](int x) {
        while (x != 0) {
            int now = x % 10;
            for (auto &&i : D) {
                if (now == i)
                    return false;
            }
            x /= 10;
        }
        return true;
    };

    constexpr int MAX = 1000000;
    for (int i = N; i < MAX; i++) {
        if (check(i)) {
            cout << i << endl;
            exit(0);
        }
    }
    return 0;
}
