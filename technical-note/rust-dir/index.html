ディレクトリ操作 in Rust

公開日：2020-05-22
更新日：-

Rust

ディレクトリが存在するか？

use std::path::Path;

if Path::new("./").is_dir() == true{
    hoge();
}

サブディレクトリを表示

use std::fs::read_dir;

let paths = read_dir("./").unwrap();
for path in paths {
    let path = path.unwrap().path();
    if path.is_dir() == true {
        println!("{:?}", path);
    }
}

full path として表示

use std::path::Path;
use std::fs;

let path = Path::new("./");
println!("{:?}", fs::canonicalize(path).unwrap());
